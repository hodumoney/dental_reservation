} // 예약 리마인더 문구 (SMS) function reminderMessage(name, dateStr) { return `[${CONFIG.HOSPITAL_NAME}] ${name}님, 내일(${dateStr}) 내원 예약이 있습니다. 변경/취소는 ${CONFIG.HOSPITAL_PHONE}로 연락 주세요.`; } // ================================================ /* ============================================================ * 공통 유틸 * ========================================================== */ function ss_() { return SpreadsheetApp.getActiveSpreadsheet(); } function tz_() { return Session.getScriptTimeZone(); } /** 시트를 얻고, 없으면 헤더와 함께 생성 */ function getApptSheet_() { const ss = ss_(); let sh = ss.getSheetByName(SHEET_APPTS); if (!sh) { sh = ss.insertSheet(SHEET_APPTS); } function jsonOut_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /** 시트 한 행 → 앱이 쓰는 객체 */ function rowToAppt_(row) { const o = {}; HEADERS.forEach((h, i) => (o[h] = row[i])); // 타입 정리 const d = o.date; o.date = (d instanceof Date) ? Utilities.formatDate(d, tz_(), "yyyy-MM-dd") : String(d || "").trim(); o.times = String(o.times || "") .split(",") .map(function (s) { return s.trim(); }) .filter(Boolean) .map(function (t) { var m = t.match(/(\d{1,2}):(\d{2})/); // "09:00" 또는 "…1899 09:00:00…" 모두 잡음 if (m) return ("0" + m[1]).slice(-2) + ":" + m[2]; return t; }); try { o.items = o.items ? } } /** POST: action 별 처리 */ function doPost(e) { const lock = LockService.getScriptLock(); lock.waitLock(20000); try { const body = JSON.parse(e.postData.contents || "{}"); if (!checkToken_(e, body)) return jsonOut_({ ok: false, error: "unauthorized" }); const action = body.action; const sh = getApptSheet_(); if (action === "add") { const a = body.appt; if (!a.id) a.id = "a" + Date.now() + Math.floor(Math.random() * 1000); // 동시간대 3명 제한 (취소 제외) if (overCapacity_(sh, a, null)) return jsonOut_({ ok: false, error: "capacity" }); if (row < 0) return jsonOut_({ ok: false, error: "notfound" }); const cur = rowToAppt_(sh.getRange(row, 1, 1, HEADERS.length).getValues()[0]); const merged = Object.assign(cur, body.patch || {}); sh.getRange(row, 1, 1, HEADERS.length).setValues([apptToRow_(merged)]); return jsonOut_({ ok: true }); } if (action === "cancel") { // 취소 = 행 삭제 대신 cancelled 플래그 (기록 보존 + 리마인더 제외) const row = findRowById_(sh, body.id); if (row < 0) return jsonOut_({ ok: false, error: "notfound" }); if (row < 0) return jsonOut_({ ok: false, error: "notfound" }); sh.deleteRow(row); return jsonOut_({ ok: true }); } if (action === "setDayStatus") { const ds = getDayStatusSheet_(); const date = body.date, status = body.status; } else { ds.appendRow([date, status]); } return jsonOut_({ ok: true }); } return jsonOut_({ ok: false, error: "unknown action" }); } catch (err) { return jsonOut_({ ok: false, error: String(err) }); } finally { lock.releaseLock(); } } /** 동시간대 3명 초과 여부 (취소 제외, self 제외) */ function overCapacity_(sh, a, selfId) { const start = [...(a.times || [])].sort()[0]; if (!start) return false; const last = sh.getLastRow(); if (last < 2) return false;
